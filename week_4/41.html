1. Selektory w jQuery

Stwórz funkcję o nazwie getPage, która pozwoli odczytać numer strony z adresu URL.
Dane na temat adresu URL strony, na której wykonuje się Twój kod JavaScript, możesz odczytać za pomocą obiektu window.location. Dopisz do adresu Twojej strony ?page=2 i wykorzystując odpowiednią właściwość obiektu location, a także odpowiednie wyrażenie regularne, zwróć z funkcji 2 (typ Number) lub null, jeśli query string ?page=X nie będzie podany lub będzie niepoprawny, np. ?page=tekst

Użycie funkcji powinno wyglądać następująco:
// dla adresu np. http://localhost/test/?page=2
getPage(); // zwraca 2
// dla adresu np. http://localhost/test/
getPage(); // zwraca null
// dla adresu np. http://localhost/test/?page=nieliczba
getPage(); // zwraca null



Rozwiązanie 

    var regex = /\/35\.html\?page=\d+$/;

    function getPage() {
    
        if (regex.test(window.location.href)) {
            return Number(/\d*$/.exec(window.location.href)[0]);
    
        } else {
            return null;
        }
    }

    var button_goToURL = document.getElementById("button_goToURL"),
        input_url = document.getElementById("input_url"),
        summary = document.getElementById("summary"),
        outputGetPage = document.getElementById("output_GetPage");
    
    var regex = /\/35\.html\?page=\d+$/;
    
    function getPage() {
    
        if (regex.test(window.location.href)) {
            return Number(/\d*$/.exec(window.location.href)[0]);
    
        } else {
            return null;
        }
    }
    
    outputGetPage.innerText = getPage();
    showSummary();
    
    button_goToURL.addEventListener("click", function () {
        window.location.href = window.location.pathname + input_url.value;
    });
    
    function showSummary() {
        summary.innerText = "adress " + window.location.href + 
                " \n cos tam cos tam ze wyrazeniem: \n /\\/35\\.html\\?page=\\d+$/";
    }
